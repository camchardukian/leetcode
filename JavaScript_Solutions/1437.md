## My Solution

```
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {boolean}
 */
var kLengthApart = function (nums, k) {
    let count = null;
    for (num of nums) {
        if (num === 1) {
            if (count === null || count >= k) {
                count = 0;
            } else if (count < k) {
                return false
            }
        } else {
            if (count === null) {
                continue;
            }
            count++;
        }
    }
    return true;
};
```
