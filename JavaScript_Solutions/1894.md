## My Solution

```
/**
 * @param {number[]} chalk
 * @param {number} k
 * @return {number}
 */
var chalkReplacer = function (chalk, k) {
    const sum = chalk.reduce((a, b) => a + b, 0);
    let remainder = k % sum;

    for (let i = 0; i < chalk.length; i++) {
        if (remainder < chalk[i]) {
            return i;
        } else if (remainder === chalk[i]) {
            remainder = 0;
            continue;
        }
        remainder -= chalk[i]
    }
};
```
