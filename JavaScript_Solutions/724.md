## My Solution

```
/**
 * @param {number[]} nums
 * @return {number}
 */
var pivotIndex = function (nums) {
    let remaining = nums.reduce((acc, cur) => acc + cur)
    let previous = 0;
    for (let i = 0; i < nums.length; i++) {
        remaining -= nums[i]
        if (previous === remaining) {
            return i;
        } else {
            previous += nums[i]
        }
    }
    return -1;
};
```
