    const stackS = []
    const stackT = []
    const maxLength = Math.max(t.length, s.length)
    for (let i = 0; i < maxLength; i++) {
        if (s[i] === undefined) {

        } else if (s[i] === "#") {
            stackS.pop()
        } else {
            stackS.push(s[i])
        }

        if (t[i] === undefined) {

        } else if (t[i] === "#") {
            stackT.pop()
        } else {
            stackT.push(t[i])
        }
    }
    return stackT.join("") === stackS.join("")
};
