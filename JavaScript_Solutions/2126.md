## My Solution

```
/**
 * @param {number} mass
 * @param {number[]} asteroids
 * @return {boolean}
 */
var asteroidsDestroyed = function (mass, asteroids) {
    asteroids.sort((a, b) => a - b);

    for (asteroid of asteroids) {
        if (asteroid > mass) {
            return false;
        }
        mass += asteroid;
    }
    return true;
};

```
