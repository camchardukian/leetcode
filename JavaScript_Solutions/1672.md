## My Solution

```
/**
 * @param {number[][]} accounts
 * @return {number}
 */
var maximumWealth = function(accounts) {
    let max = 0;
    for (let account of accounts) {
        let current = account.reduce((acc, curr) => acc + curr);
        max = Math.max(current, max);
    }
    return max;
};
```
