## My Solution

```
/**
 * @param {string} s
 * @return {number}
 */
var firstUniqChar = function(s) {
    let charMap = new Map();
    for (let char of s) {
        charMap.set(char, (charMap.get(char) || 0) + 1)
    }
    for (i = 0; i < s.length; i++) {
        if (charMap.get(s[i]) === 1) {
            return i;
        }
    }
    return -1;
};
```
