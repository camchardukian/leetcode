## My Solution

```
/**
 * @param {string} s
 * @param {number} k
 * @return {number}
 */
var maxVowels = function(s, k) {
    let maxVowels = 0;
    const vowels = /[aeiou]/

    for (let i = 0; i < k; i++) {
        if (s[i].match(vowels)) {
            maxVowels +=1
        }
    }

    let l = 0;
    let r = k;
    let current = maxVowels;
    while (r < s.length) {
        if (s[l].match(vowels)) {
            current -=1
        }
        if (s[r].match(vowels)) {
            current +=1
        }
        maxVowels = Math.max(maxVowels, current)
        l++;
        r++;
    }
    return maxVowels;
};
```
