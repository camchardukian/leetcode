## My Solution

```
/**
 * @param {number[]} candyType
 * @return {number}
 */
var distributeCandies = function (candyType) {
    const candySet = new Set();
    for (candy of candyType) {
        candySet.add(candy);
    }
    return candyType.length / 2 > candySet.size ? candySet.size : candyType.length / 2;
};
```
