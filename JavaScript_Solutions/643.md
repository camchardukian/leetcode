## My Solution

```
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var findMaxAverage = function (nums, k) {
    let greatestSum = 0;
    let l = 0;
    let r = k;
    for (let i = 0; i < k; i++) {
        greatestSum += nums[i];
    }

    let current = greatestSum;

    while (r < nums.length) {
        current -= nums[l];
        current += nums[r]
        greatestSum = Math.max(greatestSum, current)
        l++;
        r++;
    }
    return greatestSum / k;
}
```
