```
/**
 * @param {number} num
 * @return {boolean}
 */
var isPerfectSquare = function(num) {
    let min = 0;
    let max = num;
    let mid;
    while (min <= max) {
        mid = Math.floor((min + max) / 2);
        if (mid * mid === num) {
            return true;
        } else if (mid * mid > num) {
            max = mid - 1;
        } else {
            min = mid + 1;
        }
    }
    return false;
};
```
