## My Solution

```
/**
 * @param {number[][]} grid
 * @return {number}
 */
var countNegatives = function(grid) {
    let count = 0;
    for (let i = 0; i < grid.length; i++) {
        for (let ii = grid[i].length - 1; ii >= 0; ii--) {
            if (grid[i][ii] < 0) {
                count +=1;
            }
        }
    }
    return count;
};
```
