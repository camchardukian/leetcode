var sumOfEncryptedInt = function(nums) {
    let total = 0;
    for (let i = 0; i < nums.length; i++) {
    let max = 1;
    let length = 0;
    let multiplier = 10;
    while (nums[i] > 0) {
        let current = nums[i] % 10;
        max = Math.max(current, max);
        nums[i] = Math.floor(nums[i] / 10);
        multiplier*=10;
        length++;
    }
    let sum = 0;
    multiplier = 1;
    while (length > 0) {
    sum += (max * multiplier)
    length--;
      multiplier *=10;
    }
    total+=sum;
    }
    return total;
};
